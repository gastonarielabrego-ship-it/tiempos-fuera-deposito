    72→const timeToS = (t: string) => {
    73→  if (!t) return 0;
    74→  const p = t.split(':');
    75→  return (Number(p[0]) || 0) * 3600 + (Number(p[1]) || 0) * 60 + (Number(p[2]) || 0);
    76→};
    77→
    78→const formatHMS = (s: number) => {
    79→  const h = Math.floor(s / 3600);
    80→  const m = Math.floor((s % 3600) / 60);
    81→  const sec = s % 60;
    82→  return `${h}h ${m}m ${sec}s`;
    83→};
    84→const formatHM = (s: number) => {
    85→  const h = Math.floor(s / 3600);
    86→  const m = Math.floor((s % 3600) / 60);
    87→  const sec = s % 60;
    88→  return `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}:${String(sec).padStart(2, '0')}`;
    89→};

   101→const EMPRESAS_EXCLUIDAS = ['GESTION EE-EXTERNO', 'GESTION EE-EXTERNO(SELECCION)', 'G.L.D. GREMIAL EE'];
   102→// Keywords for flexible matching - if empresa contains ANY of these, it gets excluded
   103→const EMPRESAS_KEYWORDS = ['GREMIAL', 'EE-EXTERNO'];
   104→const isEmpresaExcluida = (empresa: string) => {
   105→  const upper = empresa.toUpperCase().trim();
   106→  // Direct match first
   107→  if (EMPRESAS_EXCLUIDAS.some(ex => upper.includes(ex.toUpperCase()))) return true;
   108→  // Keyword fallback
   109→  return EMPRESAS_KEYWORDS.some(kw => upper.includes(kw));
   110→};

   115→
   116→function Toast({ message, type, onClose }: { message: string; type: 'success' | 'error'; onClose: () => void }) {
   117→  useEffect(() => { const t = setTimeout(onClose, 4000); return () => clearTimeout(t); }, [onClose]);
   118→  const Icon = type === 'success' ? CheckCircle2 : XCircle;
   119→  const bg = type === 'success' ? 'bg-emerald-50 border-emerald-200 text-emerald-700' : 'bg-red-50 border-red-200 text-red-700';
   120→  return (
   121→    <div className={`fixed bottom-4 right-4 z-[100] flex items-center gap-2 px-4 py-3 rounded-lg border shadow-lg text-sm font-medium animate-in slide-in-from-bottom-2 ${bg}`}>
   122→      <Icon className="h-4 w-4 shrink-0" />
   123→      <span>{message}</span>
   124→      <button onClick={onClose} className="ml-2 opacity-50 hover:opacity-100"><XCircle className="h-3.5 w-3.5" /></button>
   125→    </div>
   126→  );
   127→}
