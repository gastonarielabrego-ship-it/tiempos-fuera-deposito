    63→    63→const timeToS = (t: string) => {
    64→    64→  if (!t) return 0;
    65→    65→  const p = t.split(':');
    66→    66→  return (Number(p[0]) || 0) * 3600 + (Number(p[1]) || 0) * 60 + (Number(p[2]) || 0);
    67→    67→};
    68→    68→
    69→    69→const formatHMS = (s: number) => {
    70→    70→  const h = Math.floor(s / 3600);
    71→    71→  const m = Math.floor((s % 3600) / 60);
    72→    72→  const sec = s % 60;
    73→    73→  return `${h}h ${m}m ${sec}s`;
    74→    74→};

   104→   104→  const bg = type === 'success' ? 'bg-emerald-50 border-emerald-200 text-emerald-700' : 'bg-red-50 border-red-200 text-red-700';
   105→   105→  return (
   106→   106→    <div className={`fixed bottom-4 right-4 z-[100] flex items-center gap-2 px-4 py-3 rounded-lg border shadow-lg text-sm font-medium animate-in slide-in-from-bottom-2 ${bg}`}>
   107→   107→      <Icon className="h-4 w-4 shrink-0" />
   108→   108→      <span>{message}</span>
   109→   109→      <button onClick={onClose} className="ml-2 opacity-50 hover:opacity-100"><XCircle className="h-3.5 w-3.5" /></button>
   110→   110→    </div>
   111→   111→  );
   112→   112→}
